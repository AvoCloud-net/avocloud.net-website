// attach to each .avo-logo element
document.querySelectorAll("[class='avo-logo']").forEach(logo => {
  const icon   = logo.querySelector("[class='avo-icon']");
  const av     = logo.querySelector("[class='avo-av']");
  const wrap   = logo.querySelector("[class='avo-text-wrap']");
  const text   = logo.querySelector("[class='avo-text']");
  const path   = logo.querySelector("[class='avo-path']");
  let open = false, busy = false;

  function expand() {
    if (open || busy) return;
    open = true;
    wrap.style.maxWidth  = "600px";
    text.style.transform = "translateX(0)";
    text.style.opacity   = "1";
    icon.style.opacity   = "0";
    icon.style.transform = "translateX(-12px)";
    av.style.opacity     = "1";
    av.style.transform   = "translateX(0)";
  }

  function collapse() {
    if (!open || busy) return;
    open = false; busy = true;
    av.style.opacity   = "0";
    av.style.transform = "translateX(8px)";
    wrap.style.maxWidth  = "0";
    text.style.transform = "translateX(20px)";
    text.style.opacity   = "0";
    setTimeout(() => {
      icon.style.transition = "none";
      icon.style.opacity    = "1";
      icon.style.transform  = "translateX(0)";
      void icon.getBoundingClientRect();
      path.style.transition      = "none";
      path.style.strokeDasharray  = "200";
      path.style.strokeDashoffset = "200";
      requestAnimationFrame(() => requestAnimationFrame(() => {
        path.style.transition      = "stroke-dashoffset 600ms cubic-bezier(.4,0,.2,1)";
        path.style.strokeDashoffset = "0";
        setTimeout(() => {
          path.style.transition = "";
          path.style.strokeDasharray = path.style.strokeDashoffset = "";
          icon.style.transition = "";
          busy = false;
        }, 630);
      }));
    }, 260);
  }

  logo.addEventListener("mouseenter", expand);
  logo.addEventListener("mouseleave", collapse);
  logo.addEventListener("focus",      expand);
  logo.addEventListener("blur",       collapse);
});
